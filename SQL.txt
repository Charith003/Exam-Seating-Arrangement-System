-- Step 3: Create students table
CREATE TABLE IF NOT EXISTS students (
    roll_number VARCHAR(20) PRIMARY KEY,  
    first_name VARCHAR(100) NOT NULL,     
    last_name VARCHAR(100) NOT NULL       
);  
-- Purpose: Create the 'students' table with columns for roll number, first name, and last name.

-- Step 4: Create rooms table
CREATE TABLE IF NOT EXISTS rooms (
    room_number VARCHAR(20) PRIMARY KEY,  
    capacity INT NOT NULL                 
);  
-- Purpose: Create the 'rooms' table with columns for room number and capacity.

-- Step 5: Create seating arrangements table
CREATE TABLE IF NOT EXISTS seating_arrangements (
    student_id VARCHAR(20),               
    room_number VARCHAR(20),              
    seat_number INT,                      
    PRIMARY KEY (student_id, room_number, seat_number),  
    FOREIGN KEY (student_id) REFERENCES students(roll_number) ON DELETE CASCADE,  
    FOREIGN KEY (room_number) REFERENCES rooms(room_number) ON DELETE CASCADE    
);  
-- Purpose: Create the 'seating_arrangements' table with foreign key references to students and rooms.
